37    /** The next request ID. */
38    private static int nextRequestId = 1;

40    /** The response handlers. */
41    private final Map<String, RegisteredResponseHandler> responseHandlers = new HashMap<String, RegisteredResponseHandler>();

43    /**
44     * Retrieve the next request ID.
45     *
46     * @return The next request ID.
47     */
48    private static String getNextRequestId() {
49      try {
50        return Long.toString(System.currentTimeMillis()) + "-" + Integer.toString(nextRequestId);
51      }
52      finally {
53        nextRequestId++;
54      }
55    }

92    /**
93     * Construct a new Request Manager.
94     *
95     * @param source The source.
96     */
97    public RequestManager(HasWebMessageHandlers source) {
98      source.addWebMessageHandler(new WebMessageHandler() {
99        @Override
100       public void onWebMessage(WebMessageEvent webMessageEvent) {
101         MessageEvent messageEvent = webMessageEvent.getMessageEvent();
102         
103         Response response = Response.asResponse(messageEvent.getData().asObject());
104         if (response != null) {
105           RequestManager.this.onResponse(response);
106         }
107       }
108     });
109   }
110   
111   /**
112    * Post a request with the given name and data to the given window with the
113    * given target origin.
114    *
115    * @param window The window.
116    * @param name The name of the request.
117    * @param data The request data.
118    * @param targetOrigin The target origin.
119    * @param responseHandler The response handler.
120    */
121   public void postRequest(Window window, String name, JsAny<?> data, String targetOrigin, final ResponseHandler responseHandler) {
122     final Request request = createRequest(name, data);
123     
124     responseHandlers.put(getRequestId(request), new RegisteredResponseHandler() {
125       @Override
126       public void onResponse(Response response) {
127         responseHandler.onResponse(request, response);
128       }
129     });
130     
131     window.postMessage(JsAny.create(request), targetOrigin);
132   }
133   
134   /**
135    * Receive notification of a response.
136    *
137    * @param response The response.
138    */
139   private void onResponse(Response response) {
140     RegisteredResponseHandler responseHandler = this.responseHandlers.remove(getRequestId(response));
141     if (responseHandler != null) {
142       responseHandler.onResponse(response);
143     }
144   }
